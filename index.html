// View Code Function
function viewCode(projectName) {
const project = projectCodes[projectName];
if (!project) return;

document.getElementById('modalTitle').textContent = project.title + ' - Code';
document.getElementById('codeLanguage').textContent = project.language;
document.getElementById('codeContent').textContent = project.code;
document.getElementById('codeModal').classList.add('active');
document.body.style.overflow = 'hidden';
}

// Close Modal Function
function closeModal() {
document.getElementById('codeModal').classList.remove('active');
document.body.style.overflow = 'auto';
}

// Copy Code Function
function copyCode() {
const codeContent = document.getElementById('codeContent').textContent;
navigator.clipboard.writeText(codeContent).then(() => {
const btn = document.querySelector('.copy-code-btn');
const originalText = btn.textContent;
btn.textContent = 'Copied!';
btn.style.background = 'var(--primary-purple)';
setTimeout(() => {
btn.textContent = originalText;
btn.style.background = 'rgba(124, 58, 237, 0.2)';
}, 2000);
});
}

// Download Code Function
function downloadCode(projectName) {
const project = projectCodes[projectName];
if (!project) return;

const blob = new Blob([project.code], {type: 'text/html' });
const url = window.URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = `${projectName}-clone.html`;
document.body.appendChild(a);
a.click();
document.body.removeChild(a);
window.URL.revokeObjectURL(url);

// Show success message
alert(`✅ ${project.title} code downloaded successfully!`);
}

// Preview Project Function
function previewProject(projectName) {
const project = projectCodes[projectName];
if (!project) return;

// Open preview in new tab
window.open(project.preview, '_blank');
}

// Initialize modal event listeners after DOM is loaded
window.addEventListener('DOMContentLoaded', function () {
// Close modal when clicking outside
const modal = document.getElementById('codeModal');
if (modal) {
modal.addEventListener('click', (e) => {
if (e.target.id === 'codeModal') {
closeModal();
}
});
}

// Close modal with Escape key
document.addEventListener('keydown', (e) => {
if (e.key === 'Escape') {
closeModal();
}
});
});
